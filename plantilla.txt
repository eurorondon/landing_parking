  // ── Particles ──
  const container = document.getElementById('particles');
  for (let i = 0; i < 20; i++) {
    const p = document.createElement('div');
    p.className = 'particle';
    const size = Math.random() * 3 + 1;
    p.style.cssText = `
      width: ${size}px; height: ${size}px;
      left: ${Math.random() * 100}%;
      background: rgba(29,143,163,${Math.random() * 0.4 + 0.1});
      animation-duration: ${Math.random() * 12 + 8}s;
      animation-delay: ${Math.random() * 8}s;
    `;
    container.appendChild(p);
  }

  // ── Scroll reveal ──
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(e => { if (e.isIntersecting) { e.target.classList.add('visible'); } });
  }, { threshold: 0.12 });
  document.querySelectorAll('.reveal').forEach(el => observer.observe(el));

  // ── Terminal selection ──
  let selectedTerminal = 'T1';
  function selectTerminal(btn) {
    document.querySelectorAll('.term-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    selectedTerminal = btn.dataset.t;
  }

  // ── FAQ ──
  function toggleFaq(btn) {
    const item = btn.closest('.faq-item');
    item.classList.toggle('open');
  }

  // ── Price calculation ──
  const RATE_PER_DAY = 8.90;
  function calcPrice() {
    const fi = document.getElementById('fecha-entrada').value;
    const fs = document.getElementById('fecha-salida').value;
    if (!fi || !fs) return { days: 0, price: 0 };
    const d1 = new Date(fi), d2 = new Date(fs);
    const diff = Math.max(1, Math.round((d2 - d1) / 86400000));
    return { days: diff, price: (diff * RATE_PER_DAY).toFixed(2) };
  }

  function fmtDate(val) {
    if (!val) return '—';
    const [y, m, d] = val.split('-');
    return `${d}/${m}/${y}`;
  }

  // ── Modal ──
  function openModal() {
    const { days, price } = calcPrice();
    document.getElementById('s-entrada').textContent = fmtDate(document.getElementById('fecha-entrada').value);
    document.getElementById('s-salida').textContent = fmtDate(document.getElementById('fecha-salida').value);
    document.getElementById('s-hora-entrada').textContent = document.getElementById('hora-entrada').value || '—';
    document.getElementById('s-hora-salida').textContent = document.getElementById('hora-salida').value || '—';
    document.getElementById('s-terminal').textContent = selectedTerminal;
    document.getElementById('s-dias').textContent = days ? `${days} día${days !== 1 ? 's' : ''}` : '—';
    document.getElementById('s-price').textContent = price ? `${price} €` : '—';
    document.getElementById('modal').classList.add('open');
    document.body.style.overflow = 'hidden';
  }

  function closeModal() {
    document.getElementById('modal').classList.remove('open');
    document.body.style.overflow = '';
  }

  document.getElementById('modal').addEventListener('click', function(e) {
    if (e.target === this) closeModal();
  });

  function confirmReservation() {
    alert('¡Reserva confirmada! Recibirás un email de confirmación en breve. ¡Buen viaje! ✈️');
    closeModal();
  }

  // ── Set default dates ──
  const today = new Date();
  const tomorrow = new Date(today);
  tomorrow.setDate(today.getDate() + 1);
  const nextWeek = new Date(today);
  nextWeek.setDate(today.getDate() + 7);
  const fmt = d => d.toISOString().split('T')[0];
  document.getElementById('fecha-entrada').value = fmt(tomorrow);
  document.getElementById('fecha-salida').value = fmt(nextWeek);
